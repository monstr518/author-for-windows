	for(x=0;x<3;++x)
	for(y=0;y<3;++y)if(!(*p)[x][y])v.push({x,y});

bool otclonenie1(A,B){
	n=0;
	for(x=0;x<3;++x)
	for(y=0;y<3;++y)if(A->game[x][y]!=B->game[x][y])++n;
	return n<=1;
}

// ѕоворот матрици по главной диагонали.
void povorot1(P){
	Q=*P;
	for(x=0;x<3;++x)
	for(y=0;y<3;++y)(*P)[x][y]=Q[y][x];
}


// ѕоворот матрици вокруг вертикали.
var verticalrepoz(P){
	for(y=0;y<3;++y){
		u=P[0][y];
		P[0][y]=P[2][y];
		P[2][y]=u;
		}
	return P;
}


// ѕоворот матрици вокруг горизонтали.
var horizontalrepoz(P){
	for(x=0;x<3;++x){
		u=P[x][0];
		P[x][0]=P[x][2];
		P[x][2]=u;
		}
	return P;
}
